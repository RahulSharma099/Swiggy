# Functional Testing Patterns Skill

## Purpose

Functional code is naturally testable. Test all layers at appropriate levels: unit tests for pure functions, integration tests for services and handlers, and end-to-end tests for complete flows.

Target coverage:

- **Services**: 80%+ (business logic)
- **Repositories**: 70%+ (data access)
- **Handlers**: 60%+ (HTTP layer - focus on integration tests)
- **Utilities**: 90%+ (pure functions)

## Unit Testing Services

Test service functions with mocked dependencies:

```typescript
// packages/api/__tests__/unit/services/issue.service.test.ts

import {
  createIssueService,
  IssueServiceDeps,
} from "@api/services/issue.service";
import { describe, it, expect, beforeEach, jest } from "@jest/globals";

describe("createIssueService", () => {
  let issueService: ReturnType<typeof createIssueService>;
  let mockDeps: IssueServiceDeps;

  beforeEach(() => {
    // Create mock dependencies
    mockDeps = {
      repos: {
        issue: {
          findByIdInWorkspace: jest.fn(),
          create: jest.fn(),
          updateWithVersionCheck: jest.fn(),
          softDelete: jest.fn(),
        },
        activity: {
          log: jest.fn(),
        },
      },
      eventPublisher: {
        emit: jest.fn(),
      },
      logger: {
        info: jest.fn(),
        warn: jest.fn(),
        error: jest.fn(),
        debug: jest.fn(),
      },
      workflowEngine: {
        canTransition: jest.fn(),
        transitionIssue: jest.fn(),
        getAvailableTransitions: jest.fn(),
      },
      authService: {
        validateWorkspaceAccess: jest.fn(),
      },
    };

    // Create service with mocked dependencies
    issueService = createIssueService(mockDeps);
  });

  describe("createIssue", () => {
    it("should create issue and emit event", async () => {
      // Arrange
      const input = {
        projectId: "p1",
        title: "New issue",
        type: "story",
        description: "Description",
      };
      const workspaceId = "ws1";
      const userId = "u1";

      const createdIssue = {
        id: "i1",
        ...input,
        creatorId: userId,
        version: 1,
        createdAt: new Date(),
      };

      jest.mocked(mockDeps.repos.issue.create).mockResolvedValue(createdIssue);

      // Act
      const result = await issueService.createIssue(input, workspaceId, userId);

      // Assert
      expect(result).toEqual(createdIssue);
      expect(mockDeps.repos.issue.create).toHaveBeenCalledWith(input, userId);
      expect(mockDeps.eventPublisher.emit).toHaveBeenCalledWith(
        "issue_created",
        expect.objectContaining({
          issueId: "i1",
          projectId: "p1",
        }),
      );
      expect(mockDeps.repos.activity.log).toHaveBeenCalled();
    });

    it("should throw ForbiddenError if user not in workspace", async () => {
      // Arrange
      jest
        .mocked(mockDeps.authService.validateWorkspaceAccess)
        .mockRejectedValue(new ForbiddenError("Not in workspace"));

      // Act & Assert
      await expect(
        issueService.createIssue(
          { projectId: "p1", title: "Test" },
          "ws1",
          "u1",
        ),
      ).rejects.toThrow(ForbiddenError);
    });

    it("should validate required fields", async () => {
      // Act & Assert
      await expect(
        issueService.createIssue(
          { projectId: "p1", title: "", type: "story" }, // Empty title
          "ws1",
          "u1",
        ),
      ).rejects.toThrow(ValidationError);
    });
  });

  describe("updateIssueStatus", () => {
    const existingIssue = {
      id: "i1",
      projectId: "p1",
      status: "todo",
      version: 1,
      assigneeId: "u2",
    };

    it("should update status and emit event", async () => {
      // Arrange
      jest
        .mocked(mockDeps.repos.issue.findByIdInWorkspace)
        .mockResolvedValue(existingIssue);
      jest
        .mocked(mockDeps.workflowEngine.canTransition)
        .mockResolvedValue(true);
      jest
        .mocked(mockDeps.repos.issue.updateWithVersionCheck)
        .mockResolvedValue({
          ...existingIssue,
          status: "in_progress",
          version: 2,
        });

      // Act
      const result = await issueService.updateIssueStatus(
        "i1",
        "in_progress",
        1,
        "ws1",
        "u1",
      );

      // Assert
      expect(result.status).toBe("in_progress");
      expect(mockDeps.eventPublisher.emit).toHaveBeenCalledWith(
        "issue_updated",
        expect.any(Object),
      );
    });

    it("should throw ConflictError on version mismatch", async () => {
      // Arrange
      jest
        .mocked(mockDeps.repos.issue.findByIdInWorkspace)
        .mockResolvedValue(existingIssue);
      jest
        .mocked(mockDeps.workflowEngine.canTransition)
        .mockResolvedValue(true);
      jest
        .mocked(mockDeps.repos.issue.updateWithVersionCheck)
        .mockResolvedValue(null); // Simulates version conflict

      // Act & Assert
      await expect(
        issueService.updateIssueStatus("i1", "in_progress", 99, "ws1", "u1"),
      ).rejects.toThrow(ConflictError);
    });

    it("should throw UnprocessableEntityError on invalid transition", async () => {
      // Arrange
      jest
        .mocked(mockDeps.repos.issue.findByIdInWorkspace)
        .mockResolvedValue(existingIssue);
      jest
        .mocked(mockDeps.workflowEngine.canTransition)
        .mockResolvedValue(false);

      // Act & Assert
      await expect(
        issueService.updateIssueStatus("i1", "invalid_status", 1, "ws1", "u1"),
      ).rejects.toThrow(UnprocessableEntityError);
    });

    it("should not emit event if update fails", async () => {
      // Arrange
      jest
        .mocked(mockDeps.repos.issue.findByIdInWorkspace)
        .mockResolvedValue(existingIssue);
      jest
        .mocked(mockDeps.workflowEngine.canTransition)
        .mockResolvedValue(true);
      jest
        .mocked(mockDeps.repos.issue.updateWithVersionCheck)
        .mockResolvedValue(null);

      // Act
      try {
        await issueService.updateIssueStatus(
          "i1",
          "in_progress",
          99,
          "ws1",
          "u1",
        );
      } catch (e) {
        // Expected to throw
      }

      // Assert
      expect(mockDeps.eventPublisher.emit).not.toHaveBeenCalled();
    });
  });
});
```

## Unit Testing Repositories

Test repository functions with a TestContainer database:

```typescript
// packages/api/__tests__/unit/repositories/issue.repository.test.ts

import { createIssueRepository } from "@api/repositories/issue.repository";
import { PrismaClient } from "@prisma/client";
import {
  describe,
  it,
  expect,
  beforeAll,
  afterAll,
  beforeEach,
} from "@jest/globals";

describe("createIssueRepository", () => {
  let prisma: PrismaClient;
  let issueRepository: ReturnType<typeof createIssueRepository>;
  let testWorkspaceId: string;
  let testProjectId: string;

  beforeAll(async () => {
    // Connect to test database
    prisma = new PrismaClient();
  });

  afterAll(async () => {
    await prisma.$disconnect();
  });

  beforeEach(async () => {
    // Create test data
    const workspace = await prisma.workspace.create({
      data: { name: "Test Workspace" },
    });
    testWorkspaceId = workspace.id;

    const project = await prisma.project.create({
      data: {
        workspaceId,
        name: "Test Project",
        description: "For testing",
      },
    });
    testProjectId = project.id;

    issueRepository = createIssueRepository(prisma, mockLogger);
  });

  afterEach(async () => {
    // Clean up test data
    await prisma.issue.deleteMany();
    await prisma.project.deleteMany();
    await prisma.workspace.deleteMany();
  });

  describe("create", () => {
    it("should create issue with all fields", async () => {
      // Act
      const result = await issueRepository.create(
        {
          title: "New Issue",
          type: "story",
          description: "Test description",
          priority: "high",
        },
        testProjectId,
        "u123",
      );

      // Assert
      expect(result).toMatchObject({
        title: "New Issue",
        type: "story",
        description: "Test description",
        priority: "high",
        projectId: testProjectId,
        creatorId: "u123",
        version: 1,
      });
      expect(result.id).toBeDefined();
      expect(result.createdAt).toBeInstanceOf(Date);
    });

    it("should create activity log entry", async () => {
      // Arrange
      const userId = "u123";

      // Act
      const issue = await issueRepository.create(
        { title: "Test" },
        testProjectId,
        userId,
      );

      // Assert
      const activity = await prisma.activityLog.findFirst({
        where: { issueId: issue.id },
      });
      expect(activity).toBeDefined();
      expect(activity?.actionType).toBe("created");
      expect(activity?.actorId).toBe(userId);
    });
  });

  describe("findByIdInWorkspace", () => {
    it("should find issue within workspace", async () => {
      // Arrange
      const issue = await prisma.issue.create({
        data: {
          projectId: testProjectId,
          title: "Test Issue",
          type: "story",
          creatorId: "u1",
        },
      });

      // Act
      const result = await issueRepository.findByIdInWorkspace(
        issue.id,
        testWorkspaceId,
      );

      // Assert
      expect(result).toBeDefined();
      expect(result?.id).toBe(issue.id);
      expect(result?.title).toBe("Test Issue");
    });

    it("should not find issue from different workspace", async () => {
      // Arrange
      const otherWorkspace = await prisma.workspace.create({
        data: { name: "Other Workspace" },
      });
      const otherProject = await prisma.project.create({
        data: {
          workspaceId: otherWorkspace.id,
          name: "Other Project",
        },
      });
      const issue = await prisma.issue.create({
        data: {
          projectId: otherProject.id,
          title: "Other Issue",
          type: "story",
          creatorId: "u1",
        },
      });

      // Act
      const result = await issueRepository.findByIdInWorkspace(
        issue.id,
        testWorkspaceId,
      );

      // Assert
      expect(result).toBeNull(); // Workspace auth check
    });
  });

  describe("updateWithVersionCheck", () => {
    it("should update with correct version", async () => {
      // Arrange
      const issue = await prisma.issue.create({
        data: {
          projectId: testProjectId,
          title: "Original",
          type: "story",
          version: 1,
          creatorId: "u1",
        },
      });

      // Act
      const result = await issueRepository.updateWithVersionCheck(
        issue.id,
        1, // Current version
        { title: "Updated" },
      );

      // Assert
      expect(result).toBeDefined();
      expect(result?.title).toBe("Updated");
      expect(result?.version).toBe(2); // Incremented
    });

    it("should reject with wrong version", async () => {
      // Arrange
      const issue = await prisma.issue.create({
        data: {
          projectId: testProjectId,
          title: "Test",
          type: "story",
          version: 5,
          creatorId: "u1",
        },
      });

      // Act
      const result = await issueRepository.updateWithVersionCheck(
        issue.id,
        1, // Wrong version
        { title: "Updated" },
      );

      // Assert
      expect(result).toBeNull(); // Failed due to version mismatch
    });
  });

  describe("searchFullText", () => {
    it("should find issues by title match", async () => {
      // Arrange
      await prisma.issue.create({
        data: {
          projectId: testProjectId,
          title: "Frontend performance issue",
          type: "bug",
          creatorId: "u1",
        },
      });

      // Act
      const results = await issueRepository.searchFullText(
        "performance",
        testProjectId,
      );

      // Assert
      expect(results.length).toBeGreaterThan(0);
      expect(results[0].title).toContain("performance");
    });
  });
});
```

```typescript
// packages/api/__tests__/integration/handlers/issue.handlers.test.ts

import request from "supertest";
import { createApp } from "@api/app";
import { describe, it, expect, beforeAll, afterAll } from "@jest/globals";

describe("Issue Handlers", () => {
  let app: Express.Application;
  let testToken: string;

  beforeAll(async () => {
    app = createApp();
    // Create test user and token
    testToken = generateTestToken("u123", "ws1");
  });

  afterAll(async () => {
    await cleanupTestData();
  });

  describe("POST /api/issues", () => {
    it("should create issue with valid input", async () => {
      // Act
      const response = await request(app)
        .post("/api/issues")
        .set("Authorization", `Bearer ${testToken}`)
        .send({
          projectId: "p1",
          title: "New feature",
          type: "story",
          description: "Add user authentication",
        });

      // Assert
      expect(response.status).toBe(201);
      expect(response.body).toMatchObject({
        success: true,
        data: {
          title: "New feature",
          type: "story",
          status: "backlog",
        },
      });
      expect(response.body.data.id).toBeDefined();
    });

    it("should reject with invalid input", async () => {
      // Act
      const response = await request(app)
        .post("/api/issues")
        .set("Authorization", `Bearer ${testToken}`)
        .send({
          projectId: "p1",
          title: "", // Empty title
          type: "invalid_type",
        });

      // Assert
      expect(response.status).toBe(400);
      expect(response.body.success).toBe(false);
      expect(response.body.code).toBe("VALIDATION_ERROR");
    });

    it("should reject without authentication", async () => {
      // Act
      const response = await request(app).post("/api/issues").send({
        projectId: "p1",
        title: "Test",
        type: "story",
      });

      // Assert
      expect(response.status).toBe(401);
      expect(response.body.code).toBe("UNAUTHORIZED");
    });
  });

  describe("PATCH /api/issues/{id}", () => {
    it("should update issue status", async () => {
      // Arrange
      const issue = await createTestIssue();

      // Act
      const response = await request(app)
        .patch(`/api/issues/${issue.id}`)
        .set("Authorization", `Bearer ${testToken}`)
        .send({
          status: "in_progress",
          version: issue.version,
        });

      // Assert
      expect(response.status).toBe(200);
      expect(response.body.data.status).toBe("in_progress");
    });

    it("should return version conflict error", async () => {
      // Arrange
      const issue = await createTestIssue();

      // Act
      const response = await request(app)
        .patch(`/api/issues/${issue.id}`)
        .set("Authorization", `Bearer ${testToken}`)
        .send({
          status: "in_progress",
          version: 99, // Wrong version
        });

      // Assert
      expect(response.status).toBe(409);
      expect(response.body.code).toBe("CONFLICT_VERSION_MISMATCH");
    });
  });
});
```

## End-to-End Testing

Test complete user flows:

```typescript
// packages/api/__tests__/e2e/issue-flow.test.ts

import request from "supertest";
import { createApp } from "@api/app";

describe("Complete Issue Flow E2E", () => {
  let app: Express.Application;
  let authToken: string;

  beforeAll(async () => {
    app = createApp();
    authToken = generateTestToken("u123", "ws1");
  });

  it("should create issue, transition status, and receive event", async () => {
    // 1. Create issue
    const createRes = await request(app)
      .post("/api/issues")
      .set("Authorization", `Bearer ${authToken}`)
      .send({
        projectId: "p1",
        title: "Implement feature X",
        type: "story",
      });

    expect(createRes.status).toBe(201);
    const issueId = createRes.body.data.id;
    const initialVersion = createRes.body.data.version;

    // 2. Transition to in_progress
    const updateRes = await request(app)
      .patch(`/api/issues/${issueId}`)
      .set("Authorization", `Bearer ${authToken}`)
      .send({
        status: "in_progress",
        version: initialVersion,
      });

    expect(updateRes.status).toBe(200);
    expect(updateRes.body.data.status).toBe("in_progress");

    // 3. Get updated issue
    const getRes = await request(app)
      .get(`/api/issues/${issueId}`)
      .set("Authorization", `Bearer ${authToken}`);

    expect(getRes.status).toBe(200);
    expect(getRes.body.data.status).toBe("in_progress");

    // 4. Check activity log
    const activityRes = await request(app)
      .get(`/api/projects/p1/activity`)
      .set("Authorization", `Bearer ${authToken}`);

    expect(activityRes.status).toBe(200);
    const issueActivityLog = activityRes.body.data.find(
      (a: any) => a.issueId === issueId,
    );
    expect(issueActivityLog).toBeDefined();
  });
});
```

## Testing Checklist

### Service Tests

- [ ] Happy path (successful operation)
- [ ] Authorization failures
- [ ] Validation failures
- [ ] Database errors (not found, conflicts)
- [ ] Dependency failures
- [ ] Event emission
- [ ] Activity logging
- [ ] Error messages are helpful
- [ ] All mocks verified called correctly

### Repository Tests

- [ ] Create operation successful
- [ ] Create activity log entry
- [ ] Find within workspace (authorization)
- [ ] Find across workspaces fails
- [ ] Update with version check succeeds
- [ ] Update with wrong version fails
- [ ] Soft delete works
- [ ] Search/filter works
- [ ] Pagination works

### Handler Tests

- [ ] Valid request returns 200/201
- [ ] Invalid request returns 400
- [ ] Missing auth returns 401
- [ ] Forbidden operation returns 403
- [ ] Not found returns 404
- [ ] Response format is standard envelope
- [ ] Meta includes timestamp
- [ ] Error response format correct

### E2E Tests

- [ ] Complete user flow works
- [ ] Data persists correctly
- [ ] Events emitted
- [ ] Audit trail created
- [ ] Cross-feature integration works

## Best Practices

✅ **DO**:

- Test pure functions with direct assertions
- Mock external dependencies (DB, API, services)
- Test error paths, not just happy paths
- Use descriptive test names
- Arrange-Act-Assert pattern
- Test at appropriate level (unit vs integration)
- Keep tests isolated (clean up after each)
- Test authorization and validation first
- Verify mocks were called correctly
- Document complex test setups

❌ **DON'T**:

- Test implementation details
- Skip error cases
- Mock too much (test integration when appropriate)
- Skip authorization/validation tests
- Use sleep() for timing (use proper async/await)
- Hardcode test data in assertions
- Test multiple concerns in one test
- Skip cleanup (causes test pollution)
- Ignore flaky tests (investigate root cause)
- Write tests you don't understand
