# WebSocket & Real-Time Development Skill

## Purpose

Real-time features keep users synchronized when data changes. The system uses Redis Pub/Sub for event broadcasting across multiple API instances, and WebSocket for client connections. All WebSocket code uses functional composition.

---

## Architecture Overview

```
API Instance 1                  API Instance 2
    ↓                               ↓
  Services                        Services
    ↓ emit events                   ↓ emit events
  Event Publisher              Event Publisher
    ↓                               ↓
    └─────────→ Redis Pub/Sub ←─────┘
                    ↓
            WebSocket Server
                    ↓
        Connected Clients (browser)
```

## Event Publisher Factory

The event publisher broadcasts events via Redis Pub/Sub.

```typescript
// services/event-publisher.ts

export interface EventPublisher {
  emit(channel: string, data: unknown): Promise<void>;
}

/**
 * Creates an event publisher using Redis Pub/Sub
 *
 * @param redis - Redis client
 * @param logger - Logger
 * @returns Event publisher instance
 */
export const createEventPublisher = (
  redis: Redis,
  logger: Logger,
): EventPublisher => ({
  async emit(channel: string, data: unknown) {
    try {
      const payload = JSON.stringify(data);
      const subscribers = await redis.publish(channel, payload);

      if (subscribers > 0) {
        logger.debug("Event published", {
          channel,
          subscribers,
          dataSize: payload.length,
        });
      }
    } catch (error) {
      logger.error("Failed to publish event", { channel, error });
      throw error;
    }
  },
});
```

## WebSocket Server Factory

```typescript
// websocket/server.ts

export interface WebSocketServer {
  broadcast(channel: string, data: unknown): void;
  sendToUser(userId: string, data: unknown): void;
  close(): Promise<void>;
}

/**
 * Creates a WebSocket server that listens to Redis events
 *
 * @param port - Port to listen on
 * @param redis - Redis instance for Pub/Sub
 * @param logger - Logger
 * @returns WebSocket server instance
 */
export const createWebSocketServer = (
  port: number,
  redis: Redis,
  logger: Logger,
): WebSocketServer => {
  const wsServer = new WebSocketServer({ port });
  const userConnections = new Map<string, Set<WebSocket>>();
  const subscriptions = new Map<string, { unsubscribe: () => Promise<void> }>();

  // Handle new connections
  wsServer.on("connection", (ws: WebSocket, req: IncomingMessage) => {
    const token = extractTokenFromUrl(req.url);
    const userId = verifyToken(token);

    if (!userId) {
      ws.close(4001, "Unauthorized");
      return;
    }

    // Track user connection
    if (!userConnections.has(userId)) {
      userConnections.set(userId, new Set());
    }
    userConnections.get(userId)!.add(ws);

    logger.info("WebSocket connected", {
      userId,
      connections: userConnections.size,
    });

    // Handle messages from client
    ws.on("message", (data: string) => {
      handleClientMessage(JSON.parse(data), userId, ws);
    });

    // Handle disconnection
    ws.on("close", () => {
      const connections = userConnections.get(userId);
      connections?.delete(ws);
      if (connections?.size === 0) {
        userConnections.delete(userId);
      }
      logger.info("WebSocket disconnected", { userId });
    });

    // Handle errors
    ws.on("error", (error) => {
      logger.error("WebSocket connection error", { userId, error });
    });
  });

  // Subscribe to Redis events
  const subscribe = async (channel: string) => {
    const subscriber = redis.duplicate();

    subscriber.on("message", (chan: string, message: string) => {
      if (chan === channel) {
        const event = JSON.parse(message);
        broadcast(channel, event);
      }
    });

    await subscriber.subscribe(channel);
    subscriptions.set(channel, {
      unsubscribe: () => subscriber.unsubscribe(channel),
    });
  };

  const broadcast = (channel: string, data: unknown) => {
    const message = JSON.stringify({
      type: "event",
      channel,
      data,
      timestamp: new Date().toISOString(),
    });

    wsServer.clients.forEach((client) => {
      if (client.readyState === WebSocket.OPEN) {
        client.send(message);
      }
    });
  };

  const sendToUser = (userId: string, data: unknown) => {
    const connections = userConnections.get(userId);
    if (!connections) return;

    const message = JSON.stringify({
      type: "direct",
      data,
      timestamp: new Date().toISOString(),
    });

    connections.forEach((ws) => {
      if (ws.readyState === WebSocket.OPEN) {
        ws.send(message);
      }
    });
  };

  return {
    broadcast,
    sendToUser,
    close: async () => {
      // Unsubscribe from all channels
      await Promise.all(
        Array.from(subscriptions.values()).map((s) => s.unsubscribe()),
      );

      // Close all client connections
      wsServer.clients.forEach((client) => {
        client.close(1000, "Server shutting down");
      });

      logger.info("WebSocket server closed");
    },
  };
};
```

## Emitting Events from Services

Services emit events that get broadcast via Redis/WebSocket:

```typescript
export const createIssueService = (deps: ServiceDeps) => {
  const { eventPublisher, logger, repos } = deps;

  return {
    async createIssue(
      input: CreateIssueInput,
      workspaceId: string,
      userId: string,
    ): Promise<Issue> {
      // Create issue...
      const issue = await repos.issue.create(input, userId);

      // Emit event for real-time sync
      await eventPublisher.emit(EventChannels.issue_created, {
        issueId: issue.id,
        projectId: issue.projectId,
        title: issue.title,
        creatorId: userId,
        createdAt: issue.createdAt.toISOString(),
      } as IssueCreatedEvent);

      logger.info("Issue creation event emitted", { issueId: issue.id });
      return issue;
    },

    async updateIssueStatus(
      issueId: string,
      newStatus: string,
      expectedVersion: number,
      workspaceId: string,
      userId: string,
    ): Promise<Issue> {
      // Update issue...
      const oldStatus = issue.status;
      const updated = await repos.issue.updateWithVersionCheck(
        issueId,
        expectedVersion,
        { status: newStatus },
      );

      // Emit both generic and specific events
      await Promise.all([
        eventPublisher.emit(EventChannels.issue_updated, {
          issueId,
          projectId: updated.projectId,
          changes: {
            status: { from: oldStatus, to: newStatus },
          },
          updatedBy: userId,
          updatedAt: new Date().toISOString(),
        } as IssueUpdatedEvent),

        eventPublisher.emit(EventChannels.issue_status_changed, {
          issueId,
          from: oldStatus,
          to: newStatus,
        }),
      ]);

      return updated;
    },
  };
};
```

## Handling Disconnections & Reconnection

```typescript
// frontend: Reconnection logic

class RealtimeConnection {
  private ws: WebSocket | null = null;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;
  private reconnectDelay = 1000;

  connect() {
    this.ws = new WebSocket("ws://localhost:3001");

    this.ws.addEventListener("open", () => {
      this.reconnectAttempts = 0;
      this.reconnectDelay = 1000;
      console.log("✓ Connected to real-time server");
      this.resubscribeToChannels();
    });

    this.ws.addEventListener("close", () => {
      console.log("✗ Disconnected from real-time server");
      this.attemptReconnect();
    });

    this.ws.addEventListener("error", (error) => {
      console.error("WebSocket error", error);
    });

    this.ws.addEventListener("message", (event) => {
      this.handleMessage(JSON.parse(event.data));
    });
  }

  private attemptReconnect() {
    if (this.reconnectAttempts >= this.maxReconnectAttempts) {
      console.error("Failed to reconnect after multiple attempts");
      return;
    }

    this.reconnectAttempts++;
    const delay = this.reconnectDelay * Math.pow(2, this.reconnectAttempts - 1);

    console.log(
      `Reconnecting in ${delay}ms (attempt ${this.reconnectAttempts})`,
    );

    setTimeout(() => this.connect(), delay);
  }

  private resubscribeToChannels() {
    // Re-subscribe to channels after reconnect
    this.ws?.send(
      JSON.stringify({
        type: "action",
        action: "resubscribe_all",
      }),
    );
  }

  private handleMessage(message: any) {
    // Handle message...
  }
}
```

## Presence Tracking

Track which users are online in each workspace:

```typescript
// websocket: Presence tracking

const userPresence = new Map<
  string,
  { userId: string; workspaceId: string; lastSeen: Date }
>();

export const setupPresenceTracking = (
  wsServer: WebSocket.Server,
  eventPublisher: EventPublisher,
) => {
  // Emit user online event
  const setUserOnline = (userId: string, workspaceId: string) => {
    userPresence.set(userId, {
      userId,
      workspaceId,
      lastSeen: new Date(),
    });

    eventPublisher.emit(EventChannels.user_online, {
      userId,
      workspaceId,
      timestamp: new Date().toISOString(),
    });
  };

  // Emit user offline event
  const setUserOffline = (userId: string, workspaceId: string) => {
    userPresence.delete(userId);

    eventPublisher.emit(EventChannels.user_offline, {
      userId,
      workspaceId,
      timestamp: new Date().toISOString(),
    });
  };

  // Get online users in workspace
  const getOnlineUsers = (workspaceId: string): string[] => {
    return Array.from(userPresence.values())
      .filter((p) => p.workspaceId === workspaceId)
      .map((p) => p.userId);
  };

  return { setUserOnline, setUserOffline, getOnlineUsers };
};
```

## Event Buffering & Replay

When clients reconnect, buffer recent events:

```typescript
// websocket: Event buffering

export const createEventBuffer = (maxSize: number = 1000) => {
  const buffer: Array<{ channel: string; data: unknown; timestamp: Date }> = [];

  return {
    add(channel: string, data: unknown) {
      buffer.push({ channel, data, timestamp: new Date() });

      // Keep buffer size limited
      if (buffer.length > maxSize) {
        buffer.shift();
      }
    },

    getRecent(channel: string, since: Date): unknown[] {
      return buffer
        .filter((e) => e.channel === channel && e.timestamp > since)
        .map((e) => e.data);
    },

    clear() {
      buffer.length = 0;
    },
  };
};

// Usage in websocket handler
const eventBuffer = createEventBuffer();

// When client connects, send buffered events
ws.on("connect", (client) => {
  const reconnectTime = client.getLastDisconnectTime();

  const bufferedEvents = eventBuffer.getRecent(
    `workspace:issue:*`,
    reconnectTime,
  );

  bufferedEvents.forEach((event) => {
    client.send(
      JSON.stringify({
        type: "buffered",
        data: event,
      }),
    );
  });
});
```

## Error Handling

```typescript
export const createErrorHandler = (logger: Logger) => ({
  handle(error: Error, context: { userId?: string; channel?: string }) {
    if (error instanceof WebSocketError) {
      logger.warn("WebSocket error", { context, error: error.message });

      if (error.code === "CLOSE_NORMAL") {
        // Normal closure, no action
      } else if (error.code === "PING_TIMEOUT") {
        // Connection lost, will reconnect
      } else {
        // Unexpected error
        throw error;
      }
    }
  },
});
```

## WebSocket Skill Checklist

- [ ] Event types defined with interfaces
- [ ] Events emitted from services after successful mutations
- [ ] WebSocket connections authenticated with JWT
- [ ] Workspace isolation enforced
- [ ] Connection tracking per user
- [ ] Graceful disconnection handling
- [ ] Reconnection with exponential backoff
- [ ] Event buffering for missed events
- [ ] Presence tracking (online/offline)
- [ ] Redis Pub/Sub for multi-instance sync
- [ ] Error logging and monitoring
- [ ] Channel naming convention followed
- [ ] Type-safe event interfaces
- [ ] No sensitive data in events
- [ ] Tests for connection/disconnection flows

---

## Key Principles

✅ **DO**:

- Emit events from services after mutations
- Use Redis Pub/Sub for multi-instance sync
- Authenticate all WebSocket connections
- Include workspace ID in all events
- Buffer recent events for reconnects
- Track user presence
- Use exponential backoff for reconnection
- Log all connection/disconnection events

❌ **DON'T**:

- Put business logic in WebSocket handlers
- Send sensitive data in events
- Forget to unsubscribe from Redis channels
- Block WebSocket event handling
- Reconnect without backoff
- Broadcast events across workspaces
- Send unbounded event payloads
